import time

def blink_colons(file_path):
    with open(file_path, 'r') as file:
        content = file.read()
    
    for char in content:
        if char == ':':
            print('\033[31m' + char + '\033[0m', end='')
            time.sleep(0.5)
            print('\033[30m' + char + '\033[0m', end='')
            time.sleep(0.5)
        else:
            print(char, end='')
    
    print()  # Print a new line after displaying the content

blink_colons('/home/dci-student/Schreibtisch/DCI-Python/240327/text_adv_bravo/_title/banner (3).txt')
